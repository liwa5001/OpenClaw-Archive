### 3. 智能评分

**改进前：**
```markdown
| 健康 | 85/100 | 固定值 ❌
| 学习 | 90/100 | 固定值 ❌
```

**改进后：**
```bash
# 健康评分计算（100 分制）
calculate_health_score() {
  # 睡眠评分（40 分）
  if (( $(echo "$sleep_val >= 7.5" | bc -l) )); then
    sleep_score=40
  else
    sleep_score=$(echo "40 - (7.5 - $sleep_val) * 10" | bc -l)
  fi
  
  # 运动评分（40 分）
  if [[ "$EXERCISE_LINE" != "无记录" ]]; then
    exercise_score=40
  else
    exercise_score=0
  fi
  
  # 体重评分（20 分）
  weight_score=20
  
  score=$((sleep_score + exercise_score + weight_score))
  echo $score
}

# 学习评分计算（100 分制）
calculate_study_score() {
  # 学习时长评分（50 分）
  time_score=$(echo "$study_min * 50 / 90" | bc)
  
  # 考题评分（50 分）
  quiz_score=$(echo "$correct * 50 / $total" | bc)
  
  score=$((time_score + quiz_score))
  echo $score
}
```
